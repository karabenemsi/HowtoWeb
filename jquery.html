$(function(){
  
  $('div p').first();
  
  $('div p').last();
  
  $('p').eq(1); //Start with 0
  
  $('p').filter('.big');
  
  $('p').not('.big');
  
});

Classes

$(function(){

  $('.button').click(function(){
    $(this).addClass('bg-pink');
  });
  
  $('.button').click(function(){
    $(this).removeClass('bg-pink');
  });

});

CSS

$(function(){

  $('.button').click(function(){
    $(this).css('borderWidth','10px');
  });
  
  $('.button').click(function(){

  });

});
